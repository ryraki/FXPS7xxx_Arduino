18uint8_t FXPS7xxx_Arduino::init() {
19 uint8_t data = 0;
20 uint8_t stat = 0;
21 if (comType == FXPS7XXX_ANA) {
22 pinMode(A0, INPUT);
23 return SENSOR_SUCCESS;
24 }
25 reset();
26 delay(100);
27 if (comType == FXPS7XXX_SPI) {
28 int i=0;
29 for (; i<100; i++) {
30 read_reg(FXPS7XXX_WHO_AM_I, 1, &data);
31 if (data==FXPS7XXX_H_WHOAMI_VALUE) break;
32 }
33 if (i==100) {
34 Serial.println("Error during initialization");
35 return FXPS7XXX_DEVICE_NOT_FOUND;
36 }
37 }
38 stat = read_reg(FXPS7XXX_WHO_AM_I, 1, &data);
39 if(data != FXPS7XXX_H_WHOAMI_VALUE){
40 Serial.println("Error during initialization");
41 return FXPS7XXX_DEVICE_NOT_FOUND;
42 }
43 read_reg(FXPS7XXX_DEVSTAT, 1, &data);
44 read_reg(FXPS7XXX_DEVSTAT, 1, &data);
45 return stat;
46}

84uint8_t FXPS7xxx_Arduino::read_reg_i2c(uint8_t address, uint8_t num, uint8_t *pReadBuffer) {
85 // In case that the user accesses the UF region
86 if (address >= 0xa0) {
87 if (((address+num)&0xf0) != (address&0xf0)) return FXPS7XXX_REGION_ERROR;
88 write_reg(address & 0xf0, FXPS7XXX_UF_REGION_W);
89 delay(1);
90 Wire.beginTransmission(I2C_address);
91 Wire.write(FXPS7XXX_UF_REGION_R);
92 Wire.endTransmission(false);
93 Wire.requestFrom(I2C_address, (uint8_t)1);
94 uint8_t data = Wire.read();
95 if (data != (address & 0xf0)) return FXPS7XXX_REGION_ERROR;
96 }
97 // Normal Operation of read function
98 Wire.beginTransmission(I2C_address);
99 Wire.write(address);
100 Wire.endTransmission(false);
101 Wire.requestFrom(I2C_address, num);
102 for (int i=0; i<num; i++)
103 pReadBuffer[i] = Wire.read();
104 return SENSOR_SUCCESS;
105}

115uint8_t FXPS7xxx_Arduino::read_reg_spi(uint8_t address, uint8_t num, uint8_t *pReadBuffer) {
116 uint8_t stat=0;
117 // In case that the user accesses the UF region
118 if (address >= 0xa0) {
119 if (((address+num)&0xf0) != (address&0xf0)) return FXPS7XXX_REGION_ERROR;
120 stat += write_reg(address & 0xf0, FXPS7XXX_UF_REGION_W);
121 delay(1);
122 uint8_t _spi_txdata_uf[4] = {FXPS7XXX_SPI_R, address, FXPS7XXX_UF_REGION_R, 0};
123 stat += access_reg_spi(_spi_txdata_uf);
124 stat += access_reg_spi(_spi_txdata_uf);
125 if (spi_rxdata[2] != (address & 0xf0)) stat = FXPS7XXX_REGION_ERROR;
126 if (stat != SENSOR_SUCCESS) return stat;
127 }
128 // Normal Operation of read function
129 uint8_t i = 0;
130 while (i<num) {
131 if (i==0 || (((address+i)%2)==0)) {
132 uint8_t _spi_txdata[4] = {FXPS7XXX_SPI_R, address+i, 0, 0};
133 stat = access_reg_spi(_spi_txdata);
134 stat += access_reg_spi(_spi_txdata);
135 }
136 pReadBuffer[i] = spi_rxdata[2-(address+i)%2];
137 i++;
138 }
139 return stat;
140}

142float FXPS7xxx_Arduino::get_pressure_sensor_data_request(uint8_t sensor) {
143 if (comType != FXPS7XXX_SPI) return -1;
144 if (sensor>=2) return -2;
145 uint8_t sourceid=0;
146 if (sensor==0) sourceid = sourceid0;
147 else sourceid = sourceid1;
148 uint8_t _spi_txdata[4] = {(sourceid<<5)|0x10, 0, 0, 0};
149 access_reg_spi(_spi_txdata);
150 uint8_t stat=access_reg_spi(_spi_txdata);
151 if (stat != SENSOR_SUCCESS) return -3;
152 uint16_t sensor_data = ((uint16_t)spi_rxdata[0]<<14) | ((uint16_t)spi_rxdata[1]<<6) | ((uint16_t)spi_rxdata[2]>>2);
153 float output;
154 if (num_sensor_data==0)
155 output=((sensor_data>>4) - PABSOFF_12sdr[P_RANGE]) / PABS_sense_12sdr[P_RANGE];
156 else
157 output=((float)sensor_data - PABSOFF_16sdr[P_RANGE]) / PABS_sense_16sdr[P_RANGE];
158 return output;
159}

161uint8_t FXPS7xxx_Arduino::write_reg_spi(uint8_t address, uint8_t writeBuffer) {
162 if (address == FXPS7XXX_SOURCEID_0)
163 sourceid0 = writeBuffer & 0x0f;
164 if (address == FXPS7XXX_SOURCEID_1)
165 sourceid1 = writeBuffer & 0x0f;
166 if (address == FXPS7XXX_SPI_CFG)
167 num_sensor_data = (writeBuffer & FXPS7XXX_SPI_CFG_DATASIZE_16)>>FXPS7XXX_SPI_CFG_DATASIZE_SHIFT;
168 uint8_t _spi_txdata[4] = {FXPS7XXX_SPI_W, address, writeBuffer, 0};
169 return access_reg_spi(_spi_txdata);
170}

172uint8_t FXPS7xxx_Arduino::access_reg_spi(uint8_t *p_spi_txdata) {
173 uint8_t buffer[4];
174 memcpy(buffer, p_spi_txdata, 4);
175 buffer[3] = u8Crc_Calculate(buffer);
176 memcpy(spi_txdata, buffer, 4);
177 digitalWrite(spi_ss, HIGH);
178 delay(1);
179 SPI.beginTransaction(spi_settings);
180 digitalWrite(spi_ss, LOW);
181 SPI.transfer(&buffer, 4);
182 digitalWrite(spi_ss, HIGH);
183 SPI.endTransaction();
184 memcpy(spi_rxdata, buffer, 4);
185 if (debug) {
186 char str[50];
187 sprintf(str, "SPI SENT: %02x %02x %02x %02x\nSPI RCVD: %02x %02x %02x %02x\n", spi_txdata[0], spi_txdata[1], spi_txdata[2], spi_txdata[3], spi_rxdata[0], spi_rxdata[1], spi_rxdata[2], spi_rxdata[3]);
188 Serial.print(str);
189 }
190 if (spi_rxdata[3] != u8Crc_Calculate(spi_rxdata)) return SENSOR_INVALIDPARAM_ERR;
191 if (spi_rxdata[0]&0x08) {
192 if (spi_rxdata[0]==0x08) return FXPS7xxx_ST_FAILED;
193 else return FXPS7XXX_DEVICE_ERROR;
194 }
195 else return SENSOR_SUCCESS;
196}

198float FXPS7xxx_Arduino::get_pressure(uint8_t sensor) {
199 if (comType == FXPS7XXX_ANA) {
200 return analogRead(A0)/1023.0*250.0+10;
201 }
202 if (sensor >= 2) return -1;
203 uint8_t data[2] = {0};
204 if (sensor == 0)
205 read_reg(FXPS7XXX_SNSDATA0_L, 2, data);
206 else
207 read_reg(FXPS7XXX_SNSDATA1_L, 2, data);
208 uint16_t raw_pressure = ((int16_t)(data[1]<< 8) | data[0]);
209 return ((float)(raw_pressure - PABSOFF_16reg[P_RANGE] ) / PABS_sense_16reg[P_RANGE]);
210}

218uint8_t FXPS7xxx_Arduino::self_test(uint8_t mode) {
219 FXPS7XXX_DSP_STAT_t dsp_status = {0};
220 uint16_t sens_16_expected = 0;
221 uint8_t stat = SENSOR_SUCCESS;
222
223 // Common Mode Self-Test
224 if (mode == FXPS7XXX_DSP_CFG_U5_STL_CTRL_P_CELL_MODE_VERIFICATION) {
225 write_reg(FXPS7XXX_DSP_CFG_U5, FXPS7XXX_DSP_CFG_U5_STL_CTRL_P_CELL_MODE_VERIFICATION);
226 delay(10);
227 write_reg(FXPS7XXX_DSP_CFG_U5, FXPS7XXX_DSP_CFG_U5_STL_CTRL_NORMAL_SIGNAL);
228 for (int i=0; i<10; i++) {
229 delay(10);
230 read_reg(FXPS7XXX_DSP_STAT, 1, &dsp_status.w);
231 if (!dsp_status.b.st_active) break;
232 }
233 if (dsp_status.b.st_active || dsp_status.b.st_error) return FXPS7xxx_ST_FAILED;
234 return SENSOR_SUCCESS;
235 }
236
237 // Fixed or Digital Test
238 else if (mode == FXPS7XXX_DSP_CFG_U5_STL_CTRL_INHIBIT_SNS_DATA_X_WRITE0 || mode == FXPS7XXX_DSP_CFG_U5_STL_CTRL_INHIBIT_SNS_DATA_X_WRITE1 || mode == FXPS7XXX_DSP_CFG_U5_STL_CTRL_INHIBIT_SNS_DATA_X_WRITE2 || mode == FXPS7XXX_DSP_CFG_U5_STL_CTRL_INHIBIT_SNS_DATA_X_WRITE3)
239 sens_16_expected = fixed_st[(mode>>4)&0x03]; // Fixed Test
240 else if (mode == FXPS7XXX_DSP_CFG_U5_STL_CTRL_DIGITAL_SELF_TEST0 || mode == FXPS7XXX_DSP_CFG_U5_STL_CTRL_DIGITAL_SELF_TEST1 || mode == FXPS7XXX_DSP_CFG_U5_STL_CTRL_DIGITAL_SELF_TEST2 || mode == FXPS7XXX_DSP_CFG_U5_STL_CTRL_DIGITAL_SELF_TEST3)
241 sens_16_expected = digital_st[(mode>>4)&0x03]; // Digital Test
242 else return FXPS7xxx_ST_FAILED;
243 write_reg(FXPS7XXX_DEVLOCK_WR, 0x00); // Make sure ENDINIT=0
244 write_reg(FXPS7XXX_DSP_CFG_U5, mode); // Enable Self-Test
245 delay(10);
246 uint8_t sens_data8[2];
247 read_reg(FXPS7XXX_SNSDATA0_L, 2, sens_data8);
248 uint16_t sens_data16 = ((int16_t)(sens_data8[1]<< 8) | sens_data8[0]);
249 if (sens_data16 != sens_16_expected) stat = FXPS7xxx_ST_FAILED;
250 write_reg(FXPS7XXX_DSP_CFG_U5, 0x00);
251 return stat;
252}

254uint8_t FXPS7xxx_Arduino::u8Crc_Calculate(uint8_t *p_u8_cmd_msg) {
255 uint32_t u32_cmd_msg = ((uint32_t)p_u8_cmd_msg[0]<<24) | ((uint32_t)p_u8_cmd_msg[1]<<16) | ((uint32_t)p_u8_cmd_msg[2]<<8);
256 struct {
257 uint8_t u8InitialValue;
258 uint16_t u16Polynomial;
259 uint8_t u8CrcLength;
260 uint8_t u8DataBitStart;
261 uint8_t u8DataLength; // including the CRC
262 } crc_calc_param = {0xFF, 0x012F, 0x08, 0, 32};
263 uint16_t u16_index;
264 /* Calculate mask value */
265 uint64_t u64_mask = ((~(uint64_t)0) >> (32 - crc_calc_param.u8DataLength)) - ((uint64_t)(((uint64_t)1) << crc_calc_param.u8CrcLength) - 1);
266 /* Update the command message using CRC params */
267 u32_cmd_msg = (u32_cmd_msg >> crc_calc_param.u8DataBitStart) & ((uint32_t)(((uint64_t)1) << crc_calc_param.u8DataLength) - 1);
268 /* Calculate remainder and XOR value */
269 uint64_t u64_remainder = (uint64_t)(((((uint64_t)u32_cmd_msg)) | (((uint64_t)crc_calc_param.u8InitialValue) << (crc_calc_param.u8DataLength))) & u64_mask);
270 uint64_t u64_xor = ((uint64_t)crc_calc_param.u16Polynomial) << (crc_calc_param.u8DataLength - 1) ;
271 /* Calculate remainder by shifting xor value over message and CRC length */
272 for (u16_index = (uint16_t)
273 (((uint16_t)crc_calc_param.u8DataLength) +
274 ((uint16_t)crc_calc_param.u8CrcLength)); u16_index > 0;
275 u16_index--) {
276 if (((u64_remainder >> (u16_index - 1)) & 0x01) != 0)
277 u64_remainder ^= u64_xor;
278 u64_xor >>= 1;
279 if ((u64_remainder & u64_mask) == 0)
280 break;
281 }
282 /* Return 8-bit CRC calculated */
283 uint8_t CRC_Answer = (uint8_t)(u64_remainder & (uint8_t)((uint64_t)(((uint64_t)1) << crc_calc_param.u8CrcLength) - 1));
284 //p_u8_cmd_msg[3] = CRC_Answer;
285 return CRC_Answer;
286}
